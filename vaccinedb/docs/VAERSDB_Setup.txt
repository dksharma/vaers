DROP DATABASE IF EXISTS `vaersdb`;
CREATE DATABASE IF NOT EXISTS `vaersdb`; 
USE `vaersdb`;

DROP TABLE IF EXISTS `vaccine`;
CREATE TABLE IF NOT EXISTS `vaccine` (
  `VAERS_ID` int(10) NOT NULL,
  `VAX_TYPE` varchar(50) NOT NULL,
  `VAX_MANU` varchar(256) NOT NULL,
  `VAX_LOT` varchar(256) DEFAULT NULL,
  `VAX_DOSE` varchar(32) DEFAULT NULL,
  `VAX_ROUTE` varchar(32) DEFAULT NULL,
  `VAX_SITE` varchar(32) DEFAULT NULL,
  `VAX_NAME` varchar(2048) DEFAULT NULL
) ENGINE=InnoDB DEFAULT CHARSET=latin1 COMMENT='vaccine table';

DROP TABLE IF EXISTS `symptoms`;
CREATE TABLE IF NOT EXISTS `symptoms` (
  `VAERS_ID` int(10) NOT NULL,
  `SYMPTOM1` varchar(1024) DEFAULT NULL,
  `SYMPTOMVERSION1` varchar(256) DEFAULT NULL,
  `SYMPTOM2` varchar(1024) DEFAULT NULL,
  `SYMPTOMVERSION2` varchar(256) DEFAULT NULL,
  `SYMPTOM3` varchar(1024) DEFAULT NULL,
  `SYMPTOMVERSION3` varchar(256) DEFAULT NULL,
  `SYMPTOM4` varchar(1024) DEFAULT NULL,
  `SYMPTOMVERSION4` varchar(256) DEFAULT NULL,
  `SYMPTOM5` varchar(1024) DEFAULT NULL,
  `SYMPTOMVERSION5` varchar(256) DEFAULT NULL
) ENGINE=InnoDB DEFAULT CHARSET=latin1 COMMENT='Vaccine Symptoms';

DROP TABLE IF EXISTS `vaersdata`;
CREATE TABLE IF NOT EXISTS `vaersdata` (
  `VAERS_ID` int(10) NOT NULL,
  `RECVDATE` varchar(50) DEFAULT NULL,
  `STATE` varchar(32) DEFAULT NULL,
  `AGE_YRS` varchar(50) DEFAULT NULL,
  `CAGE_YR` varchar(50) DEFAULT NULL,
  `CAGE_MO` varchar(50) DEFAULT NULL,
  `SEX` varchar(32) DEFAULT NULL,
  `RPT_DATE` varchar(50) DEFAULT NULL,
  `SYMPTOM_TEXT` varchar(4096) DEFAULT NULL,
  `DIED` varchar(32) DEFAULT NULL,
  `DATEDIED` varchar(50) DEFAULT NULL,
  `L_THREAT` varchar(32) DEFAULT NULL,
  `ER_VISIT` varchar(32) DEFAULT NULL,
  `HOSPITAL` varchar(32) DEFAULT NULL,
  `HOSPDAYS` varchar(32) DEFAULT NULL,
  `X_STAY` varchar(32) DEFAULT NULL,
  `DISABLE` varchar(32) DEFAULT NULL,
  `RECOVD` varchar(32) DEFAULT NULL,
  `VAX_DATE` varchar(50) DEFAULT NULL,
  `ONSET_DATE` varchar(50) DEFAULT NULL,
  `NUMDAYS` varchar(50) DEFAULT NULL,
  `LAB_DATA` varchar(4096) DEFAULT NULL,
  `V_ADMINBY` varchar(256) DEFAULT NULL,
  `V_FUNDBY` varchar(256) DEFAULT NULL,
  `OTHER_MEDS` varchar(1024) DEFAULT NULL,
  `CUR_ILL` varchar(4096) DEFAULT NULL,
  `HISTORY` varchar(4096) DEFAULT NULL,
  `PRIOR_VAX` varchar(256) DEFAULT NULL,
  `SPLTTYPE` varchar(256) DEFAULT NULL
) ENGINE=InnoDB DEFAULT CHARSET=latin1 COMMENT='data ';

DROP TABLE IF EXISTS `vaccinenondomestic`;
CREATE TABLE IF NOT EXISTS `vaccinenondomestic` (
  `VAERS_ID` int(10) NOT NULL,
  `VAX_TYPE` varchar(50) NOT NULL,
  `VAX_MANU` varchar(256) NOT NULL,

  `VAX_ROUTE` varchar(32) DEFAULT NULL,

DROP TABLE IF EXISTS `symptomsnondomestic`;

  `SYMPTOM4` varchar(1024) DEFAULT NULL,
  `SYMPTOMVERSION4` varchar(256) DEFAULT NULL,
  `SYMPTOM5` varchar(1024) DEFAULT NULL,
  `SYMPTOMVERSION5` varchar(256) DEFAULT NULL
) ENGINE=InnoDB DEFAULT CHARSET=latin1 COMMENT='Vaccine Symptoms for non domestic data';

DROP TABLE IF EXISTS `vaersdatanondomestic`;

  `CAGE_YR` varchar(50) DEFAULT NULL,
  `CAGE_MO` varchar(50) DEFAULT NULL,
  `SEX` varchar(32) DEFAULT NULL,
  `RPT_DATE` varchar(50) DEFAULT NULL,
  `SYMPTOM_TEXT` varchar(4096) DEFAULT NULL,
  `DIED` varchar(32) DEFAULT NULL,
  `DATEDIED` varchar(50) DEFAULT NULL,
  `L_THREAT` varchar(32) DEFAULT NULL,
  `ER_VISIT` varchar(32) DEFAULT NULL,

  `HOSPDAYS` varchar(32) DEFAULT NULL,

  `VAX_DATE` varchar(50) DEFAULT NULL,
  `ONSET_DATE` varchar(50) DEFAULT NULL,
  `NUMDAYS` varchar(50) DEFAULT NULL,
  `LAB_DATA` varchar(4096) DEFAULT NULL,
  `V_ADMINBY` varchar(256) DEFAULT NULL,
  `V_FUNDBY` varchar(256) DEFAULT NULL,
  `OTHER_MEDS` varchar(1024) DEFAULT NULL,
  `CUR_ILL` varchar(4096) DEFAULT NULL,

  `PRIOR_VAX` varchar(256) DEFAULT NULL,
  `SPLTTYPE` varchar(256) DEFAULT NULL
